``` ts
class SMJoinState implements ISMState {
    // ...ISMState implementation
    private received: Map<SMTransitionId, SMStateStartEvent>;

    get isComplete(): boolean {
        this.received.length === this.incoming.length;
    }

    constuctor(id: SMStateId)  {
        // ...
    }

    onDependencyComplete(evt: SMStateStartEvent) {
        this.received[evt.transitionId] = evt;
    }

    reset() {
        this.received.clear();
    }

    get receivedPayloads() : SMStateStartEvent[] {
        return Objects.values(this.received);
    }
}
```

```ts
type Handler<T> = (event: T) => void;

class TypedEvent<T> {
  private handlers: Handler<T>[] = [];

  public add(handler: Handler<T>) {
    this.handlers.push(handler);
  }

  public emit(event: T) {
    this.handlers.forEach(h => h(event));
  }
}
```

```ts
interface ISMStateMachine {
    id: SMStateMachineId;

    onSMStarted: TypedEvent<SMStartedEvent>
    onStateStart: TypedEvent<SMStateStartEvent>
    onStateStopped: TypedEvent<SMStateStoppedEvent>
    onSMStopped: TypedEvent<SMStoppedEvent>

    start();
    stop();

    onStopped(id: ISMStateId, status: SMStatus, exitCode?: string, payload?: Map<string, any>);
    
    addState(state: ISMState);
    removeState(id: SMStateId);
    getState(id: SMStateId) : ISMState | undefined;
    getStateCount(): number;
    getStateIds(): ReadonlyArray<SMStateId>;
    getActiveStateIds(): ReadonlyArray<SMStateId>;

    addTransition(transition: ISMTransition);
    removeTransition(id: SMTransitionId);
    getState(id: SMTransitionId) : ISMTransition | undefined;
    getTransitionCount(): number;
    getTransitionIds() : ReadonlyArray<SMTransitionId>;

    createInitial(id: SMStateId, payload: Map<string, any>) : ISMState;
    createState(id: SMStateId, config: Map<string, any>) : ISMState;
    createTerminal(id: SMStateId) : ISMState;
    createChoice(id: SMStateId) : ISMState;
    createFork(id: SMStateId) : ISMState;
    createJoin(id: SMStateId) : SMJoinState;
    createGroup(id: SMStateId) : SMGroupState;
}
```
